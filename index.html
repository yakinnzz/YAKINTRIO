🐷 1. POLIMORFISME - Hewan DART

Polimorfisme adalah kemampuan suatu objek untuk memiliki banyak bentuk. Dalam contoh ini, variabel bertipe Hewan dapat merujuk ke objek Babi.

// Nama : USMAN YAKIN TRIO SINADA
// NIM  : 251410006
// Kelas: SI2A

class Hewan {
  void bersuara() => print('...');
}

class Babi extends Hewan {
  @override
  void bersuara() => print('grok-grok');
}

void main() {
  Hewan hewan = Babi();
  hewan.bersuara(); // Output: grok-grok
}

🚗 5. ABSTRACT CLASS - Kendaraan DART

Abstract class adalah kelas yang tidak bisa diinstansiasi langsung. Method start() harus diimplementasikan oleh kelas turunannya.

// Nama : USMAN YAKIN TRIO SINADA
// NIM  : 251410006
// Kelas: SI2A

abstract class Kendaraan {
  String? merk;
  void start();
  void stop() => print('Kendaraan berhenti');
}

class Mobil extends Kendaraan {
  @override
  void start() => print('Mobil $merk menyala');
}

void main() {
  Mobil m = Mobil();
  m.merk = "SUPRA";
  m.start(); // Mobil SUPRA menyala
  m.stop();  // Kendaraan berhenti
}

📱 10. MULTIPLE INHERITANCE - Tablet DART

Dart mendukung multiple inheritance melalui interface (implements). Tablet mengimplementasikan 3 interface: Komputer, Kamera, Telepon.

// Nama : USMAN YAKIN TRIO SINADA
// Nim : 251410006
// Kelas : SI2A

abstract class Bentuk {
  double hitungLuas();
  double hitungKeliling();

  void info() {
    print('Luas: ${hitungLuas()} , Keliling: ${hitungKeliling()}');
  }
}

class Persegi  implements Bentuk {
  double sisi;
  Persegi(this.sisi);

  @override
  double hitungLuas() => sisi * sisi;

  @override
  double hitungKeliling() => 4 * sisi;

  @override
  void info() {
    print(
      'Persegi dengan sisi $sisi memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}'
    );
  }
}

class Lingkaran implements Bentuk {
  double radius;
  Lingkaran(this.radius);

  @override
  double hitungLuas() => 3.14 * radius * radius;

  @override
  double hitungKeliling() => 2 * 3.14 * radius;

  @override
  void info() {
    print(
      'Lingkaran dengan radius $radius memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}'
    );
  }
}

void main() {
  Bentuk b1 = Persegi(4);
  Bentuk b2 = Lingkaran(7);

  b1.info();
  b2.info();
}
}

import 'dart:math';

class MathUtils {
  static int faktorial(int n) {
    if (n <= 1) return 1;
    return n * faktorial(n - 1);
  }

  static bool isPrima(int n) {
    if (n < 2) return false;

    for (int i = 2; i <= sqrt(n); i++) {
      if (n % i == 0) return false;
    }
    return true;
  }

  static double roundDouble(double value, int decimal) {
    double mod = pow(10, decimal).toDouble();
    return (value * mod).round() / mod;
  }

  static int randomInt(int min, int max) {
    return min + Random().nextInt(max - min + 1);
  }

  static double randomDouble(double min, double max) {
    return min + Random().nextDouble() * (max - min);
  }

  static double rataRata(List data) {
    return data.reduce((a, b) => a + b) / data.length;
  }

  static double median(List data) {
    var sorted = List.from(data);
    sorted.sort();
    int n = sorted.length;

    if (n % 2 == 1) {
      return sorted[n ~/ 2].toDouble();
    } else {
      return (sorted[n ~/ 2 - 1] + sorted[n ~/ 2]) / 2;
    }
  }
}

📱 14.Media Player - Tablet DART

Dart mendukung Media Player melalui interface (implements). Tablet mengimplementasikan 3 interface: Komputer, Kamera, Telepon.

// Nama : USMAN YAKIN TRIO SINADA
// NIM : 251410006
// Kelas : SI2A

abstract class MediaPlayer {
  void load(String file);
}

abstract class Playable {
  void play();
}

abstract class Pausable {
  void pause();
}

class AudioPlayer extends MediaPlayer implements Playable, Pausable {
  String? file;

  @override
  void load(String file) {
    this.file = file;
    print('Audio dimuat: $file');
  }

  @override
  void play() {
    print('Audio diputar');
  }

  @override
  void pause() {
    print('Audio dijeda');
  }
}

class VideoPlayer extends MediaPlayer implements Playable, Pausable {
  String? file;

  @override
  void load(String file) {
    this.file = file;
    print('Video dimuat: $file');
  }

  @override
  void play() {
    print('Video diputar');
  }

  @override
  void pause() {
    print('Video dijeda');
  }
}

void main() {
  List players = [
    AudioPlayer(),
    VideoPlayer(),
  ];

  for (var p in players) {
    p.load('file.mp3 / file.mp4');

    if (p is Playable) {
      p.play();
    }

    if (p is Pausable) {
      p.pause();
    }

    print('---');
  }
}
